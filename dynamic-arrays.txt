#include<iostream>
using namespace std;

int main()
{
int r,c;
int **p;//double pointer
while(1){
	cout << "Enter rows:";
	cin >> r;
	cout <<"Enter columns:";
	cin >>c;
p = new int*[r];//array of pointers
for(int i=0;i<r;i++)
	p[i] = new int[c];//dynamic array creation

for(int i=0;i<r;i++)
{
	for(int j=0;j<c;j++)
		cout << p[i][j] << "\t";
cout << endl;
}
delete p;
}
	return 0;
}
