#include<iostream>
using namespace std;

int main()
{
int list[7] = {45,5,6,7,5,9,11};
int temp,swaps=0,searches=0,smallest=0;
int location;
for (int i = 1; i < 7;i++){//start from 2nd item to end
            if (list[i] < list[i - 1]) //compare the two items
			{
			  searches++;
               temp = list[i]; //store item and location
               location = i;
               do //compare present item to all the sorted items and insert in right place
               {
					swaps++;
					list[location] = list[location - 1];
                     location--;
               }while (location > 0 && list[location - 1] > temp);
               list[location] = temp;
      } 

}


for(int i=0;i<7;i++)
	cout << list[i] << "\t";
cout << endl;
cout << "Total moves = " << swaps << endl;
cout << "Total searches = " << searches << endl;
return 0;
}
