#include<iostream>
using namespace std;

int main()
{
int list[7] = {3, 5, 7, 9, 13, 47, 66};
int searches=0,num=0,match=0;
int first=0,last=6,mid;
cout << "Enter a number to search for:";
cin >> num;

for(int i=0;i<3;i++)//log2n value
{
mid=(first+last)/2;
if(num==list[mid])
{
	match=1;
	break;
}
else if (num<list[mid])
	last=mid-1;
else
	first=mid+1;
}

if(match==1)
	cout << "Item found" << endl;
else
	cout << "Item not found" << endl;

return 0;
}
